    public static void Suma(params int[] liczby){
      
     for(int i=0; i< liczby.Length;i++){
            Console.Write(liczby[i]+ 10+", ");
        }  
    }
    public static void Main(string[] args)
    {
        int[] cyfry = new int[3]{1,2,3};
